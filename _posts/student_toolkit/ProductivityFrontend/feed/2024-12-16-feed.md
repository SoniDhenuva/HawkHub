<script>
  // Storage key
  const STORAGE_KEY = 'feedItems';

  // Get DOM elements
  const uploadArea = document.getElementById('uploadArea');
  const uploadBtn = document.getElementById('uploadBtn');
  const fileInput = document.getElementById('fileInput');
  const feedItemsContainer = document.getElementById('feedItems');

  // Click to upload
  uploadBtn.addEventListener('click', () => fileInput.click());

  // Drag and drop
  uploadArea.addEventListener('dragover', (e) => {
    e.preventDefault();
    uploadArea.style.background = 'var(--bg-2)';
    uploadArea.style.borderColor = 'var(--accent-bright)';
  });

  uploadArea.addEventListener('dragleave', () => {
    uploadArea.style.background = 'var(--bg-1)';
    uploadArea.style.borderColor = 'var(--accent)';
  });

  uploadArea.addEventListener('drop', (e) => {
    e.preventDefault();
    uploadArea.style.background = 'var(--bg-1)';
    uploadArea.style.borderColor = 'var(--accent)';
    handleFiles(e.dataTransfer.files);
  });

  // File input change
  fileInput.addEventListener('change', (e) => {
    handleFiles(e.target.files);
  });

  function handleFiles(files) {
    const feedItems = loadFeedItems();
    
    for (let file of files) {
      const reader = new FileReader();
      reader.onload = (e) => {
        const item = {
          id: Date.now() + Math.random(),
          fileName: file.name,
          fileSize: file.size,
          fileType: file.type,
          uploadedAt: new Date().toLocaleString(),
          fileData: e.target.result // Base64 encoded file
        };
        
        feedItems.unshift(item);
        saveFeedItems(feedItems);
        renderFeedItems(feedItems);
      };
      reader.readAsDataURL(file);
    }
    
    // Reset file input
    fileInput.value = '';
  }

  function saveFeedItems(items) {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(items));
  }

  function loadFeedItems() {
    const items = localStorage.getItem(STORAGE_KEY);
    return items ? JSON.parse(items) : [];
  }

  function renderFeedItems(items) {
    if (items.length === 0) {
      feedItemsContainer.innerHTML = `
        <div class="empty-state">
          <div class="empty-state-icon">📭</div>
          <p>No files uploaded yet. Start by uploading your first file!</p>
        </div>
      `;
      return;
    }

    feedItemsContainer.innerHTML = items.map(item => {
      const fileIcon = getFileIcon(item.fileType);
      const fileSize = (item.fileSize / 1024).toFixed(2) + ' KB';
      
      return `
        <div class="feed-item">
          <div class="feed-item-header">
            <h3 class="feed-item-title">${item.fileName}</h3>
            <button class="download-btn" onclick="downloadFile('${item.id}')">Download</button>
          </div>
          <div class="feed-item-meta">
            ${fileSize} • Uploaded ${item.uploadedAt}
          </div>
          <div class="feed-item-file">
            <span class="file-icon">${fileIcon}</span>
            <span class="file-name">${item.fileName}</span>
            <button onclick="deleteFile('${item.id}')" style="background: #ef4444; color: white; border: none; padding: 0.5rem 0.75rem; border-radius: 6px; cursor: pointer; font-size: 0.85rem;">Delete</button>
          </div>
        </div>
      `;
    }).join('');
  }

  function getFileIcon(fileType) {
    if (fileType.startsWith('image/')) return '🖼️';
    if (fileType.startsWith('video/')) return '🎥';
    if (fileType.startsWith('audio/')) return '🎵';
    if (fileType.includes('pdf')) return '📄';
    if (fileType.includes('word')) return '📝';
    if (fileType.includes('sheet')) return '📊';
    return '📎';
  }

  function downloadFile(itemId) {
    const items = loadFeedItems();
    const item = items.find(i => i.id == itemId);
    
    if (item) {
      const link = document.createElement('a');
      link.href = item.fileData;
      link.download = item.fileName;
      link.click();
    }
  }

  function deleteFile(itemId) {
    if (confirm('Are you sure you want to delete this file?')) {
      let items = loadFeedItems();
      items = items.filter(i => i.id != itemId);
      saveFeedItems(items);
      renderFeedItems(items);
    }
  }

  // Initial render
  renderFeedItems(loadFeedItems());
</script>
